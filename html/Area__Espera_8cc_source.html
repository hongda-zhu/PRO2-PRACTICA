74 void Area_Espera::enviar_procesos_cluster(const int& n, Cluster &c) {
75  /*
76  Aplicamos dos bucles while para hacer la iteración:
77  La primera iteración nos sirve para actualizar los numeros de envios aceptados y de envios rechazados.
78  La segunda iteración nos sirve para comprobar el estado del envío tal que cuando el estado del envío es true, quiere decir que hemos dado de alta correctamente el proceso que ha enviado, sino volvemos a insertar el proceso como si fuese nuevo a la misma prioridad;
79  */
80 
81  /*
82  Inv:
83  i = el contador de los procesos aceptados respect el n
84  j = el contador de los procesos existentes de la prioridad que se está tratando
85  it = la posición de las prioridades que está situado en el parametro implicito, que comienza desde el begin() hasta el end() en el caso extremo
86  size = la cantidad del procesos que tiene en una prioridad;
87  info_prioridad = Pair<int n_aceptados, int n_rechazados>
88  info_prioridad.first = los numeros totales de procesos aceptados de la prioridad tratada
89  info_prioridad.second = los numeros totales de procesos rechazados de la prioridad tratada
90  0 <= i <= n
91  0 <= j <= size
92  */
93  int i = 0;
94  map<string,Prioridad>::iterator it = prioridades_data.begin();
95  // Cota : n - i e dist(it, prioridades_data.end())
96  while (i < n and it != prioridades_data.end()) {
97  int j = 0;
98  int size = it -> second.retrieve_jobs_size();
99  pair<int, int> info_prioridad = it -> second.retrieve_info();
100  while (i < n and j < size) {
101  // Devolvemos el proceso más antiguo de la prioridad
102  Proceso temp = it -> second.retrieve_job();
103  bool is_sent = c.enviar_procesos_cluster(temp);
104  // is_sent == true cuando el proceso se ha enviado correctamente
105  if (is_sent) {
106  ++i;
107  ++info_prioridad.first;
108  }
109  else {
110  // si is_sent == false, entonces el proceso no se ha enviado correctamente, volvemos a agregar el proceso a la area de espera
111  it -> second.alta_proceso_espera(temp);
112  ++info_prioridad.second;
113  }
114  ++j;
115  }
116  /*
117  Post :
118  1. (i == n): se ha enviado correctamente n procesos
119  2. (j == size): todos los procesos de esta prioridad ha sido enviado una vez
120  */
121  it -> second.update_prioridad(info_prioridad.first,info_prioridad.second);
122  ++it;
123  }
124  /*
125  Post:
126  1. (i == n): se ha enviado correctamente n procesos
127  2. (it == prioridades.end()): los procesos de todas las prioridades del parametro implicito han sido tratados
128  */
129 }

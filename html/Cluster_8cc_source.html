68 bool Cluster::i_modificar_cluster(BinTree<string> &prcd, map<string, Procesador>&prcd_data, const string &id_prcd, const Cluster &c2) {
69  // Busca la posición de procesador que queremos modificar en la estructura de procesadores
70  // res = result, result es falso cuando el procesador que queremos modificar tiene procesadores auxiliares y es true cuando hemos cambiado su valor correctamente
71  bool res = false;
72  // caso base 1: si estamos apuntando un nodo vacio sin procesador
73  if (prcd.empty()) res = false;
74  // caso base 2 : si encontramos el procesador y tal procesador contiene un procesador auxiliar, entonces devolvemos false
75  else if (prcd.value() == id_prcd and (not (prcd.left().empty() and prcd.right().empty()))) res = false;
76  // caso base 3: si hemos pasado al caso uno, esto quiere decir el id que hemos encontrado es una hoja, entonces eliminamos el procesador que queremos consultar y añadimos el nuevo procesador;
77  else if (prcd.value() == id_prcd) {
78  // modificamos el nodo en el árbol original
79  prcd = c2.prcd;
80  // insertamos los elementos nuevo del nodo que queremos modificar al prcd_data
81  prcd_data.erase(id_prcd);
82  prcd_data.insert(c2.prcd_data.begin(), c2.prcd_data.end());
83  res = true;
84  } else {
85  // left
86  BinTree<string>left = prcd.left();
87  bool is_left = i_modificar_cluster(left, prcd_data, id_prcd, c2);
88  // si hemos encontrado el nodo por la parte izquierda
89  if (is_left) {
90  // modificamos el árbol
91  prcd = BinTree<string>(prcd.value(), left, prcd.right());
92  res = true;
93  } else {
94  // right
95  BinTree<string>right = prcd.right();
96  bool is_right = i_modificar_cluster(right, prcd_data, id_prcd, c2);
97  // si hemos encontrado el nodo por por la parte derecha
98  if (is_right) {
99  prcd = BinTree<string>(prcd.value(), prcd.left(), right);
100  res = true;
101  }
102  }
103  }
104  return res;
105 }

181 void Cluster::i_enviar_procesos_cluster (const Proceso &job, const BinTree<string> &prcd, string &id_prcd, int &free_memory, int &gap, int &depth, int counter) {
182  // llamada recursiva en forma preorden cumpliendo los criterios mencionados
183  // El caso base sería cuando el nodo estamos tratando no es un nodo vacío
184  if (not prcd.empty()) {
185  string temp_id_prcd = prcd.value();
186  map<string,Procesador>::iterator it = prcd_data.find(temp_id_prcd);
187  bool exist_job = it -> second.exist_job(job.retrieve_id());
188  if (not exist_job){
189  // 1. comparamos el hueco del procesador que estamos tratando es más ajustado respecto con el procesador que hemos registrado
190  if (it -> second.fit_job(job.retrieve_size())) {
191  int temp_free_memory = it -> second.retrieve_free_memory();
192  int temp_gap = it -> second.best_fit(job.retrieve_size());
193  if (gap == 0 or gap > temp_gap) {
194  // 1.1 si cumple la condición, sustituimos el procesador registrado respecto al procesador que estamos tratando
195  gap = temp_gap;
196  free_memory = temp_free_memory;
197  id_prcd = temp_id_prcd;
198  depth = counter;
199  } else if (temp_gap == gap) {
200  // 2. chequeamos quien tiene mayor memoria libre
201  if (temp_free_memory > free_memory) {
202  free_memory = temp_free_memory;
203  id_prcd = temp_id_prcd;
204  depth = counter;
205  } else if (temp_free_memory == free_memory) {
206  // 3. si tiene la misma memoria libre, chequeamos quien tiene está más cercano al nodo principal
207  if (depth > counter) {
208  id_prcd = temp_id_prcd;
209  depth = counter;
210  }
211  }
212  }
213  }
214  // al recorrer la estructura en preorden y tener los criterios anteriores cubiertos, el criterio de "procesador de más izquierda" se cumple automaticamente
215  }
216 
217  // llamada recursiva en preorden, incrementando el contador cada vez +1
218  i_enviar_procesos_cluster(job, prcd.left(), id_prcd, free_memory, gap, depth, ++counter);
219  i_enviar_procesos_cluster(job, prcd.right(), id_prcd, free_memory, gap, depth, ++counter);
220  }
221 }
